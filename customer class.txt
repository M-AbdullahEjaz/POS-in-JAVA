public class customer {
    public int customerId;
    public String name;
    public String address;
    public String phone;
    public String email;
    public double amountPayable;
    public double salesLimit;

    public customer(int id,String name,String address,String phone,String email, double balance,double limit)
    {
      this.customerId=id;
      this.name=name;
      this.address=address;
      this.phone=phone;
      this.email=email;
      this.amountPayable=balance;
      this.salesLimit=limit;
    }
    public String toFileString()
    {
        return customerId+";"+name+";"+address+";"+phone+";"+email+";"+amountPayable+";"+salesLimit+";";
    }
    public static customer fromFileString(String line)
    {
        String[] parts=line.split(";");
        return new customer(Integer.parseInt(parts[0]),parts[1],parts[2],parts[3],parts[4],Double.parseDouble(parts[5]),Double.parseDouble(parts[6]));
    }
}
